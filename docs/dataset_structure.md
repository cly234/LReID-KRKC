```
├── market1501

│   │── bounding_box_train
│   └── ...
├── cuhksysu4reid
│   │── combine

│   └── ...
├── MSMT17
│   │── test
│   │── train
│   └── ...
├── viper
│   │── VIPeR
│   │   │── cam_a

```
